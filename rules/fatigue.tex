\begin{itemize}
  \item
  Every strenuous activity inflicts one \gls{fatigue}.
  \begin{itemize}
    \item
    Climbing
    \item
    Combat
    \item
    Swimming
  \end{itemize}
  \item
  Every \gls{fatigue} a character has above their current \glspl{hp} inflicts a -1 penalty to all actions.
\end{itemize}
