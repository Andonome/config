\AddToHook{shipout/background}{%
  \ifodd\value{page}
  \shiftColours[\value{segNoSpare}]
  \color{background}
  \stepcounter{segNoSpare}
    \def\BGpageanchor{current page.south east}
    \def\BGadjust{-5mm, 5mm}
  \else
    \def\BGpageanchor{current page.south west}
    \def\BGadjust{5mm, 5mm}
  \fi
  \begin{tikzpicture}[overlay,remember picture]
    \ifnum\value{page}=10
      \draw[gray, dotted] (-0.15,0) -- (-0.15,-10);
    \fi
    \ifnum\value{page}=14
      \draw[gray, dotted] (-0.15,0) -- (-0.15,-10);
      \draw[gray, dotted] (0,0) -- (5,0);
    \fi
    \path 
      ++(\BGpageanchor)
      +(\BGadjust)
      node[
        text centered,
        text=black,
        font=\titleFont,
      ] {\thepage};
  \end{tikzpicture}%
}%
