\newcommand\showPage{%
  \iftoggle{countedPage}{
    \ifnum\thepage>12%
      \ifodd\thepage%
        \node at ([xshift=-2em]interior.south east) {\thepage} ;%
      \else%
        \node at ([xshift=2em]interior.south west) {\thepage} ;%
      \fi%
    \else%
      \ifnum\thepage<6%
        \ifodd\thepage%
          \node at ([xshift=-2em]interior.south east) {\thepage} ;%
        \else%
          \node at ([xshift=2em]interior.south west) {\thepage} ;%
        \fi%
      \else%
        \ifodd\thepage%
          \node[rotate=180] at ([xshift=24em]interior.south) {\thepage} ;%
        \else%
          \node[rotate=-180] at ([xshift=2em]interior.south west) {\thepage} ;%
        \fi%
      \fi%
    \fi%
  }{}%
}
